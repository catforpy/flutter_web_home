## 📐 重构原则

### 1. SOLID 原则

**S - 单一职责原则 (Single Responsibility)**
```dart
// ❌ 错误：一个类做太多事情
class PageEditor {
  List<Map<String, dynamic>> carouselImages;
  List<Map<String, dynamic>> searchImages;
  List<Map<String, dynamic>> videoImages;
  // ... 14+ 种组件的数据
  // 加上编辑逻辑
  // 加上渲染逻辑
  // 加上配置逻辑
}

// ✅ 正确：每个类只负责一件事
class PageEditor {
  final List<PageComponent> components;
}

class CarouselComponent {
  final List<CarouselImage> images;
  final CarouselStyle style;
}

class CarouselStyle {
  final bool autoPlay;
  final Duration duration;
  // ...
}
```

**O - 开闭原则 (Open/Closed)**
```dart
// ❌ 错误：添加新组件需要修改核心类
class ComponentPreviewContent {
  Widget buildPreview(String componentName) {
    switch (componentName) {
      case '轮播': return _buildCarousel();
      case '搜索': return _buildSearch();
      // 每次添加新组件都要修改这里
    }
  }
}

// ✅ 正确：通过扩展添加新功能
abstract class PageComponent {
  Widget buildPreview(BuildContext context);
  Widget buildEditor(BuildContext context);
}

class CarouselComponent extends PageComponent {
  @override
  Widget buildPreview(BuildContext context) { /* ... */ }

  @override
  Widget buildEditor(BuildContext context) { /* ... */ }
}
```

**L - 里氏替换原则 (Liskov Substitution)**
```dart
// 所有组件可以互相替换使用
List<PageComponent> components = [
  CarouselComponent(),
  VideoComponent(),
  NoticeComponent(),
];
```

**I - 接口隔离原则 (Interface Segregation)**
```dart
// ❌ 错误：接口过大
abstract class Component {
  Widget buildPreview();
  Widget buildEditor();
  void validateImages();
  void validateText();
  void validateLinks();
  void uploadMedia();
  void deleteMedia();
}

// ✅ 正确：接口分离
abstract class Component {
  Widget buildPreview();
  Widget buildEditor();
}

abstract class MediaUploadable {
  Future<void> uploadMedia();
}

abstract class Validatable {
  ValidationResult validate();
}
```

**D - 依赖倒置原则 (Dependency Inversion)**
```dart
// ❌ 错误：依赖具体实现
class InstructorManagementPage {
  void loadData() {
    _instructors = MockInstructorData.load(); // 硬编码
  }
}

// ✅ 正确：依赖抽象
class InstructorManagementPage extends StatelessWidget {
  final InstructorRepository repository;

  const InstructorManagementPage({
    required this.repository,
  });
}
```

/// 页面组件基类
abstract class PageComponent {
  final String id;
  final String name;
  final ComponentType type;
  final bool enabled;
  final int sortOrder;

  const PageComponent({
    required this.id,
    required this.name,
    required this.type,
    this.enabled = true,
    this.sortOrder = 0,
  });

  /// 构建预览
  Widget buildPreview(BuildContext context);

  /// 构建编辑器
  Widget buildEditor(BuildContext context);

  /// 序列化为 JSON
  Map<String, dynamic> toJson();

  /// 从 JSON 创建
  static PageComponent fromJson(Map<String, dynamic> json) {
    final type = ComponentType.values.firstWhere(
      (e) => e.name == json['type'],
      orElse: () => ComponentType.unknown,
    );

    switch (type) {
      case ComponentType.carousel:
        return CarouselComponent.fromJson(json);
      case ComponentType.video:
        return VideoComponent.fromJson(json);
      case ComponentType.notice:
        return NoticeComponent.fromJson(json);
      // ... 其他组件
      default:
        throw UnimplementedError('Component type $type not implemented');
    }
  }
}

/// 组件类型枚举
enum ComponentType {
  carousel,        // 轮播
  search,          // 搜索
  categoryNav,     // 分类导航
  video,           // 视频
  notice,          // 公告
  title,           // 标题
  image,           // 图片
  showcase,        // 橱窗
  button,          // 按钮
  divider,         // 分割线
  imageTextList,   // 图文列表
  courseList,      // 课程列表
  recommendList,   // 推荐列表
  teacherList,     // 讲师列表
  unknown,         // 未知
}

/// 轮播组件
class CarouselComponent extends PageComponent {
  final List<CarouselImage> images;
  final CarouselStyle style;

  const CarouselComponent({
    required super.id,
    required this.images,
    required this.style,
    super.enabled = true,
    super.sortOrder = 0,
  }) : super(
          name: '轮播',
          type: ComponentType.carousel,
        );

  @override
  Widget buildPreview(BuildContext context) {
    return CarouselPreviewWidget(component: this);
  }

  @override
  Widget buildEditor(BuildContext context) {
    return CarouselEditorWidget(component: this);
  }

  @override
  Map<String, dynamic> toJson() => {
        'id': id,
        'type': type.name,
        'enabled': enabled,
        'sortOrder': sortOrder,
        'images': images.map((e) => e.toJson()).toList(),
        'style': style.toJson(),
      };

  factory CarouselComponent.fromJson(Map<String, dynamic> json) {
    return CarouselComponent(
      id: json['id'] as String,
      images: (json['images'] as List)
          .map((e) => CarouselImage.fromJson(e as Map<String, dynamic>))
          .toList(),
      style: CarouselStyle.fromJson(json['style'] as Map<String, dynamic>),
      enabled: json['enabled'] as bool? ?? true,
      sortOrder: json['sortOrder'] as int? ?? 0,
    );
  }

  CarouselComponent copyWith({
    List<CarouselImage>? images,
    CarouselStyle? style,
    bool? enabled,
    int? sortOrder,
  }) {
    return CarouselComponent(
      id: id,
      images: images ?? this.images,
      style: style ?? this.style,
      enabled: enabled ?? this.enabled,
      sortOrder: sortOrder ?? this.sortOrder,
    );
  }
}

/// 轮播样式
class CarouselStyle {
  final bool autoPlay;
  final Duration duration;
  final Duration interval;
  final Color indicatorColor;
  final Color activeIndicatorColor;
  final double height;
  final EdgeInsets margin;
  final EdgeInsets padding;
  final double borderRadius;

  const CarouselStyle({
    this.autoPlay = true,
    this.duration = const Duration(milliseconds: 1000),
    this.interval = const Duration(milliseconds: 4000),
    this.indicatorColor = Colors.white,
    this.activeIndicatorColor = const Color(0xFF1A9B8E),
    this.height = 200,
    this.margin = EdgeInsets.zero,
    this.padding = EdgeInsets.zero,
    this.borderRadius = 0,
  });

  Map<String, dynamic> toJson() => {
        'autoPlay': autoPlay,
        'duration': duration.inMilliseconds,
        'interval': interval.inMilliseconds,
        'indicatorColor': '#${indicatorColor.value.toRadixString(16)}',
        'activeIndicatorColor': '#${activeIndicatorColor.value.toRadixString(16)}',
        'height': height,
        'marginTop': margin.top,
        'marginBottom': margin.bottom,
        'paddingLeft': padding.left,
        'paddingRight': padding.right,
        'borderRadius': borderRadius,
      };

  factory CarouselStyle.fromJson(Map<String, dynamic> json) {
    return CarouselStyle(
      autoPlay: json['autoPlay'] as bool? ?? true,
      duration: Duration(milliseconds: json['duration'] as int? ?? 1000),
      interval: Duration(milliseconds: json['interval'] as int? ?? 4000),
      indicatorColor: _parseColor(json['indicatorColor'] as String?),
      activeIndicatorColor: _parseColor(json['activeIndicatorColor'] as String?),
      height: (json['height'] as num?)?.toDouble() ?? 200,
      margin: EdgeInsets.only(
        top: (json['marginTop'] as num?)?.toDouble() ?? 0,
        bottom: (json['marginBottom'] as num?)?.toDouble() ?? 0,
      ),
      padding: EdgeInsets.only(
        left: (json['paddingLeft'] as num?)?.toDouble() ?? 0,
        right: (json['paddingRight'] as num?)?.toDouble() ?? 0,
      ),
      borderRadius: (json['borderRadius'] as num?)?.toDouble() ?? 0,
    );
  }

  static Color? _parseColor(String? colorString) {
    if (colorString == null || colorString.isEmpty) return null;
    return Color(int.parse(colorString.replaceFirst('#', '0xFF')));
  }

  CarouselStyle copyWith({
    bool? autoPlay,
    Duration? duration,
    Duration? interval,
    Color? indicatorColor,
    Color? activeIndicatorColor,
    double? height,
    EdgeInsets? margin,
    EdgeInsets? padding,
    double? borderRadius,
  }) {
    return CarouselStyle(
      autoPlay: autoPlay ?? this.autoPlay,
      duration: duration ?? this.duration,
      interval: interval ?? this.interval,

class InstructorBloc extends Bloc<InstructorEvent, InstructorState> {
  final InstructorRepository repository;

  InstructorBloc({required this.repository}) : super(InstructorInitial()) {
    on<LoadInstructors>(_onLoadInstructors);
    on<SearchInstructors>(_onSearchInstructors);
    on<FilterInstructors>(_onFilterInstructors);
    on<SortInstructors>(_onSortInstructors);
    on<DeleteInstructor>(_onDeleteInstructor);
    on<UpdateInstructorStatus>(_onUpdateInstructorStatus);
  }

  Future<void> _onLoadInstructors(
    LoadInstructors event,
    Emitter<InstructorState> emit,
  ) async {
    emit(InstructorLoading());
    final result = await repository.getInstructors(
      page: event.page,
      pageSize: event.pageSize,
    );
    result.fold(
      (failure) => emit(InstructorError(failure.message)),
      (instructors) => emit(InstructorLoaded(
        instructors: instructors,
        currentPage: event.page,
        hasMore: instructors.length == event.pageSize,
      )),
    );
  }

  Future<void> _onSearchInstructors(
    SearchInstructors event,
    Emitter<InstructorState> emit,
  ) async {
    if (state is InstructorLoaded) {
      emit((state as InstructorLoaded).copyWith(isSearching: true));
      final result = await repository.searchInstructors(event.keyword);
      result.fold(
        (failure) => emit(InstructorError(failure.message)),
        (instructors) => emit(InstructorLoaded(
          instructors: instructors,
          currentPage: 1,
          searchKeyword: event.keyword,
        )),
      );
    }
  }

  // ... 其他事件处理
}

class FormDialog<T> extends StatelessWidget {
  final String title;
  final List<FormField> fields;
  final FormSubmitCallback<T> onSubmit;
  final String submitButtonText;

  const FormDialog({
    super.key,
    required this.title,
    required this.fields,
    required this.onSubmit,
    this.submitButtonText = '提交',
  });

  @override
  Widget build(BuildContext context) {
    return AlertDialog(
      title: Text(title),
      content: SingleChildScrollView(
        child: Form(
          key: _formKey,
          child: Column(
            mainAxisSize: MainAxisSize.min,
            children: fields,
          ),
        ),
      ),
      actions: [
        TextButton(
          onPressed: () => Navigator.pop(context),
          child: const Text('取消'),
        ),
        ElevatedButton(
          onPressed: _handleSubmit,
          child: Text(submitButtonText),
        ),
      ],
    );
  }

  Future<void> _handleSubmit() async {
    if (_formKey.currentState?.validate() ?? false) {
      final result = await onSubmit(_getFormData());
      if (context.mounted) {
        Navigator.pop(context, result);
      }
    }
  }
}
